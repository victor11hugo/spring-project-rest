CREATE TABLE TB_MOVIE ( ID_MOVIE NUMBER NOT NULL,
MV_NAME VARCHAR2(100),
MV_GENRE VARCHAR2(20),
MV_DATE VARCHAR2(50),
ID_DIRECTOR NUMBER,
ID_RATING NUMBER,
CONSTRAINT PK_MOVIE PRIMARY KEY (ID_MOVIE),
CONSTRAINT FK_DIRECTOR FOREIGN KEY (ID_DIRECTOR) REFERENCES TB_DIRECTOR(ID_DIRECTOR),
CONSTRAINT FK_RATING FOREIGN KEY (ID_RATING) REFERENCES TB_RATING(ID_RATING));

CREATE TABLE TB_DIRECTOR ( ID_DIRECTOR NUMBER NOT NULL,
DR_NAME VARCHAR2(100),
CONSTRAINT PK_DIRECTOR PRIMARY KEY (ID_DIRECTOR) );

CREATE TABLE TB_RATING ( ID_RATING NUMBER NOT NULL,
RT_VALUE NUMBER,
CONSTRAINT PK_RATING PRIMARY KEY (ID_RATING) );

CREATE SEQUENCE MOVIE_SEQ
INCREMENT BY 1
START WITH 1
MAXVALUE 99999999999
MINVALUE 1;

CREATE SEQUENCE DIRECTOR_SEQ
INCREMENT BY 1
START WITH 1
MAXVALUE 99999999999
MINVALUE 1;

CREATE SEQUENCE RATING_SEQ
INCREMENT BY 1
START WITH 1
MAXVALUE 99999999999
MINVALUE 1;
